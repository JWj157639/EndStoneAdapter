### 核心前提：吃透`SetFocus`官方规则（方案不失效的关键）
从文档提炼3个**强制规则**（之前未提及，必须遵守），否则`SetFocus`会直接返回NULL失败：
1. **线程绑定要求**：目标窗口（WT）必须**附加到调用线程的消息队列**，跨线程直接调用`SetFocus`必失败（核心限制）；
2. **窗口状态要求**：目标窗口不能是**禁用状态**（`IsWindowEnabled`返回TRUE），否则报错`ERROR_INVALID_PARAMETER (0x57)`；
3. **返回值校验**：成功返回**原焦点窗口HWND**，失败返回NULL，**必须通过返回值判断是否成功**，而非仅看`GetLastError`。

### 完整方案：`SendInput+SetFocus+AttachThreadInput`（合规版）
新增`AttachThreadInput`API解决**线程绑定**核心限制，其余逻辑保留，全程4步，**原有SendInput代码仍无改动**，仅新增线程绑定/解绑逻辑，完美符合官方文档要求。

#### 方案核心逻辑
1. 用固定特征定位目标WT窗口HWND，校验窗口有效性；
2. 调用`AttachThreadInput`将**当前程序线程**与**目标窗口线程**绑定，让目标窗口附加到当前线程消息队列（解决SetFocus核心限制）；
3. 调用`SetFocus`绑定输入焦点到目标窗口，严格校验返回值；
4. 调用`SendInput`模拟键鼠，完成后**解绑线程**（必做，避免系统输入异常）。

### 一、完整C++代码（可直接编译运行）
适配你的场景，包含**窗口枚举、线程绑定、焦点设置、键鼠模拟、资源释放**全流程，注释详尽，直接替换特征即可使用。
```cpp
#include <Windows.h>
#include <tchar.h>
#include <stdio.h>

// ************************* 你的目标窗口固定特征（写死，永久不变） *************************
const TCHAR* TARGET_TITLE = _T("d:\\mengli\\bedrock_server_mod.exe"); // WT窗口标题
const TCHAR* TARGET_CLASS = _T("CASCADIA_HOSTING_WINDOW_CLASS");     // WT固定类名（x64dbg验证后替换）
// **************************************************************************************

HWND g_hTargetWnd = NULL; // 存储匹配到的目标窗口HWND

// 窗口枚举回调函数：按标题+类名筛选目标窗口
BOOL CALLBACK EnumWindowsProc(HWND hwnd, LPARAM lParam)
{
    TCHAR szTitle[512] = { 0 };
    TCHAR szClass[512] = { 0 };
    GetWindowText(hwnd, szTitle, _countof(szTitle)); // 获取窗口标题
    GetClassName(hwnd, szClass, _countof(szClass)); // 获取窗口类名

    // 匹配条件：标题包含目标路径 + 类名完全匹配 + 窗口有效且启用
    if (_tcsstr(szTitle, TARGET_TITLE) != NULL &&
        _tcscmp(szClass, TARGET_CLASS) == 0 &&
        IsWindow(hwnd) && IsWindowEnabled(hwnd) && IsWindowVisible(hwnd))
    {
        g_hTargetWnd = hwnd;
        return FALSE; // 找到目标，终止枚举
    }
    return TRUE; // 继续枚举
}

// 函数：获取目标窗口HWND（按固定特征枚举）
HWND GetTargetHWND()
{
    g_hTargetWnd = NULL;
    EnumWindows(EnumWindowsProc, 0); // 枚举系统所有窗口
    if (g_hTargetWnd == NULL)
    {
        printf("❌ 未找到目标WindowsTerminal窗口，请检查特征是否正确！\n");
    }
    else
    {
        printf("✅ 找到目标窗口，HWND：0x%X\n", (UINT)g_hTargetWnd);
    }
    return g_hTargetWnd;
}

// 函数：模拟键鼠操作（原有SendInput代码，无任何改动）
// 示例：模拟按下ESC键（可替换为任意按键/组合键/鼠标操作）
BOOL SimulateInput()
{
    INPUT input[2] = { 0 }; // 2个事件：ESC按下 + ESC松开
    // 1. ESC键按下
    input[0].type = INPUT_KEYBOARD;
    input[0].ki.wVk = VK_ESCAPE; // ESC虚拟键码
    input[0].ki.dwFlags = 0;     // 0=按键按下
    // 2. ESC键松开
    input[1].type = INPUT_KEYBOARD;
    input[1].ki.wVk = VK_ESCAPE;
    input[1].ki.dwFlags = KEYEVENTF_KEYUP; // 按键松开标志

    // 调用SendInput注入输入
    UINT uRet = SendInput(_countof(input), input, sizeof(INPUT));
    if (uRet != _countof(input))
    {
        printf("❌ SendInput模拟失败，错误码：%d，成功注入：%d/%d\n", GetLastError(), uRet, _countof(input));
        return FALSE;
    }
    printf("✅ 键鼠模拟成功！已向目标窗口发送ESC键\n");
    return TRUE;
}

int _tmain(int argc, TCHAR* argv[])
{
    // 步骤1：获取目标窗口HWND
    HWND hTarget = GetTargetHWND();
    if (hTarget == NULL) return -1;

    // 步骤2：获取目标窗口线程ID和当前程序线程ID
    DWORD dwTargetThreadId = GetWindowThreadProcessId(hTarget, NULL); // 目标窗口线程ID
    DWORD dwCurrentThreadId = GetCurrentThreadId();                   // 当前程序线程ID
    if (dwTargetThreadId == dwCurrentThreadId)
    {
        printf("✅ 目标窗口与当前程序同线程，无需绑定\n");
    }
    else
    {
        // 绑定线程：让目标窗口附加到当前线程消息队列（解决SetFocus线程限制）
        if (!AttachThreadInput(dwCurrentThreadId, dwTargetThreadId, TRUE))
        {
            printf("❌ AttachThreadInput线程绑定失败，错误码：%d\n", GetLastError());
            return -1;
        }
        printf("✅ 线程绑定成功（当前线程：%d → 目标窗口线程：%d）\n", dwCurrentThreadId, dwTargetThreadId);
    }

    // 步骤3：调用SetFocus设置输入焦点到目标窗口（严格遵循官方文档）
    HWND hOldFocus = SetFocus(hTarget);
    if (hOldFocus == NULL)
    {
        printf("❌ SetFocus焦点设置失败，错误码：%d\n", GetLastError());
        // 失败后解绑线程（若已绑定）
        if (dwTargetThreadId != dwCurrentThreadId)
        {
            AttachThreadInput(dwCurrentThreadId, dwTargetThreadId, FALSE);
        }
        return -1;
    }
    printf("✅ 输入焦点设置成功！原焦点窗口HWND：0x%X\n", (UINT)hOldFocus);
    Sleep(50); // 给系统50ms处理焦点绑定，避免输入失效

    // 步骤4：执行SendInput键鼠模拟（原有代码，无改动）
    SimulateInput();

    // 步骤5：资源释放（必做！避免系统输入线程混乱）
    if (dwTargetThreadId != dwCurrentThreadId)
    {
        // 解绑线程：恢复系统原有线程输入关系
        if (AttachThreadInput(dwCurrentThreadId, dwTargetThreadId, FALSE))
        {
            printf("✅ 线程解绑成功\n");
        }
        else
        {
            printf("❌ 线程解绑失败，错误码：%d\n", GetLastError());
        }
    }
    // 可选：恢复原焦点窗口（友好交互）
    SetFocus(hOldFocus);
    printf("✅ 程序执行完成，所有资源已释放\n");

    system("pause");
    return 0;
}
```

### 二、关键编译&运行要求
1. **编译环境**：Visual Studio/MinGW均可，无需额外库（链接`User32.lib`，VS自动链接，MinGW加`-luser32`）；
2. **权限要求**：**管理员身份运行**（右键→以管理员身份运行），否则`AttachThreadInput`/`SetFocus`会权限不足；
3. **类名校验**：用x64dbg附加WT进程，执行`getclassname 窗口HWND`获取真实类名，替换代码中`TARGET_CLASS`；
4. **窗口状态**：确保WT窗口（bedrock）处于**运行、启用、非最小化**状态（最小化可加`ShowWindow(hTarget, SW_SHOWNOACTIVATE)`唤醒）。

### 三、核心API调用细节（严格遵循官方文档）
#### 1. `AttachThreadInput`（线程绑定，解决SetFocus核心限制）
```cpp
AttachThreadInput(当前线程ID, 目标窗口线程ID, TRUE); // 绑定
AttachThreadInput(当前线程ID, 目标窗口线程ID, FALSE); // 解绑（必做）
```
- 作用：让两个线程共享输入状态，目标窗口会附加到当前线程消息队列，满足`SetFocus`的**线程绑定要求**；
- 必解绑定：模拟完成后必须解绑，否则会导致系统键盘/鼠标输入异常（如焦点无法切换）。

#### 2. `SetFocus`（输入焦点绑定，严格校验）
```cpp
HWND hOldFocus = SetFocus(hTarget);
if (hOldFocus == NULL) { /* 失败处理 */ }
```
- 官方返回值规则：成功返回**原焦点窗口HWND**，失败返回NULL，**不可用`hOldFocus == hTarget`判断**；
- 失败原因：窗口禁用、线程未绑定、权限不足，对应错误码可通过`GetLastError`排查。

#### 3. `SendInput`（原有代码无改动）
- 可直接替换为任意键鼠模拟逻辑（组合键、鼠标点击、移动等），输入会**精准流向已绑定焦点的目标窗口**；
- 保留返回值校验：返回值为成功注入的事件数，需等于INPUT数组长度才表示完全成功。

### 四、常见问题&避坑指南（对应官方文档）
| 问题现象 | 原因 | 解决方案 |
|----------|------|----------|
| `SetFocus`返回NULL，错误码0 | 目标窗口与当前程序跨线程，未调用`AttachThreadInput` | 先执行线程绑定，再调用`SetFocus` |
| `SetFocus`报错`0x57`（ERROR_INVALID_PARAMETER） | 目标窗口处于**禁用状态**（`IsWindowEnabled`返回FALSE） | 确保WT窗口未被禁用，代码中加`IsWindowEnabled`校验 |
| `AttachThreadInput`失败，错误码5 | 权限不足 | 以**管理员身份**运行程序 |
| 模拟输入生效，但系统键鼠焦点混乱 | 未调用`AttachThreadInput(FALSE)`解绑线程 | 模拟完成后**必须解绑线程**，即使中间步骤失败 |
| 未找到目标窗口（g_hTargetWnd=NULL） | 标题/类名匹配错误，或WT窗口未运行 | 用x64dbg重新验证WT窗口的标题和类名，确保窗口正在运行 |
